# 🚀 Advanced Fetch API

```javascript
const API_URL =
  "https://jsonplaceholder.typicode.com/users";

async function getUsers() {
  try {
    const response = await fetch(API_URL);

    if (!response.ok) {
      throw new Error("API Error");
    }

    const data = await response.json();

    console.table(data);

  } catch (error) {
    console.error(error.message);
  }
}

getUsers();
```
